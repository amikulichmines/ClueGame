To set doorways, we had:

String cellInfo = tempGrid.get(r)[c];
if(cellInfo.length() > 1) {
	if(cellInfo.length() > 1) {
		char addInfo = cellInfo.charAt(1);
		else if(addInfo == '<') {
			cell.setDoorway(true);
			cell.setDoorDirection(addInfo);
		}
		else if(addInfo == '^') {
			cell.setDoorway(true);
			cell.setDoorDirection(addInfo);
		}
		else if(addInfo == '>') {
			cell.setDoorway(true);
			cell.setDoorDirection(addInfo);
		}
		else if(addInfo == 'v') {
			cell.setDoorway(true);
			cell.setDoorDirection(addInfo);
		}
	}
}

Instead of 4 if-statements, we can do:


String cellInfo = tempGrid.get(r)[c];
if(cellInfo.length() > 1) {
	if(cellInfo.length() > 1) {
		char addInfo = cellInfo.charAt(1);
		if("^v<>".contains(""+addInfo)) {
			cell.setDoorway(true);
			cell.setDoorDirection(addInfo);
		}
	}
}
